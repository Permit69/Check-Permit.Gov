<script>
    function checkData() {
        const country = document.getElementById('countrySelect').value;
        const id = document.getElementById('permitID').value.trim();
        const resultDiv = document.getElementById('result');

        if (!country || !id) {
            alert("Please select a country and enter a number!");
            return;
        }

        resultDiv.style.display = "block";
        resultDiv.innerHTML = "Checking database...";
        resultDiv.className = "";

        setTimeout(() => {
            // সিম্পল লজিক: আপনি এখানে আপনার ডাটাবেস বা নাম্বার সেট করতে পারেন
            if (id === "12345") {
                resultDiv.className = "success";
                resultDiv.innerHTML = `<strong>Status: APPROVED</strong><br>Country: ${country}<br>Your permit is verified.`;
            } else {
                resultDiv.className = "error";
                resultDiv.innerHTML = `<strong>Status: NOT FOUND</strong><br>No data for ${country} with this ID.`;
            }
        }, 1000);
    }
</script>
