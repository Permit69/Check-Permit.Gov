    <script>
        function checkStatus() {
            const country = document.getElementById('country').value;
            const number = document.getElementById('permitNumber').value.trim();
            const resultBox = document.getElementById('result');

            if (country === "" || number === "") {
                alert("Please select a country and enter a valid number!");
                return;
            }

            // লোডিং ইফেক্ট
            resultBox.style.display = "block";
            resultBox.className = "";
            resultBox.innerHTML = "Searching in " + country + " database...";

            setTimeout(() => {
                // এখানে আপনার রেজাল্ট লজিক (উদাহরণস্বরূপ ১০০০১ দিলে অ্যাপ্রুভ দেখাবে)
                if (number === "10001" || number === "786") {
                    resultBox.className = "success";
                    resultBox.innerHTML = "<strong>Status: APPROVED!</strong><br>Country: " + country + "<br>Your permit is verified and ready.";
                } else {
                    resultBox.className = "error";
                    resultBox.innerHTML = "<strong>Status: NOT FOUND!</strong><br>We couldn't find any data for " + number + " in " + country + ".";
                }
            }, 1500);
        }
    </script>
